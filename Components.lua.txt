				pulseBusy = false
			end)

		if input.UserInputType == Enum.UserInputType.MouseMovement then
			pulse()
		end
	end)

function PrestineLib:AddParagraphBars(params)
	local tabName = params.Tab
	local mainTitle = params.MainTitle
	local mainContent = params.MainContent

	local theme = params.Theme or {}
	local accentColor = theme.AccentColor or Color3.fromRGB(110, 160, 235)
	local glowColor = theme.GlowColor or Color3.fromRGB(100, 150, 230)
	local strokeColor = theme.StrokeColor or Color3.fromRGB(90, 140, 220)
	local bgColor = theme.BackgroundColor or Color3.fromRGB(14, 16, 26)
	local completeColor = theme.CompleteColor or Color3.fromRGB(0, 200, 120)

	local bgTransparency = theme.BackgroundTransparency or 0.62
	local strokeTransparency = theme.StrokeTransparency or 0.78
	local strokeThickness = theme.StrokeThickness or 2

	local compact = (params.Compact == true)

	local wrapWidth = params.Width or 325
	local paddingX = params.PaddingX or 12
	local topPad = params.TopPad or 8
	local titleHeight = params.TitleHeight or 22
	local dividerY = params.DividerY or 30
	local contentY = params.ContentY or 38
	local showContent = (params.ShowContent ~= false)
	local contentTextSize = params.ContentTextSize or (compact and 12 or 13)

	local bars = params.Bars or {}

	local barHeight = params.BarHeight or (compact and 7 or 8)
	local barSpacing = params.BarSpacing or (compact and 6 or 8)
	local labelTextSize = params.LabelTextSize or (compact and 11 or 12)
	local countTextSize = params.CountTextSize or (compact and 11 or 12)
	local rowDivider = (params.ShowRowDivider ~= false)
	local rowDividerTransparency = params.RowDividerTransparency or 0.92

	local segmentsDefault = math.clamp(math.floor((params.Segments or (compact and 10 or 12)) + 0.5), 4, 40)

	local headerLineHeight = params.HeaderLineHeight or (compact and 14 or 16)
	local rowHeight = params.BarRowHeight or (headerLineHeight + barHeight + 4)

	local microDelay = params.MicroDelay
	if microDelay == nil then microDelay = 0.05 end
	microDelay = math.clamp(microDelay, 0, 0.2)

	local contentHeight
	if params.ContentHeight ~= nil then
		contentHeight = params.ContentHeight
	else
		contentHeight = showContent and (compact and 16 or 18) or 0
	end

	local function computeHeight(n)
		local startY = contentY + (showContent and contentHeight or 0) + 6
		local rowsHeight = (n > 0) and (n * rowHeight + (n - 1) * barSpacing) or 0
		local bottomPad = params.BottomPad or 10
		return startY + rowsHeight + bottomPad
	end

	local paragraphSize = params.paragraphSize or computeHeight(#bars)

	local targetTab = Main:FindFirstChild(tabName .. "Contents")
	if not targetTab then return end

	local mainScroller = targetTab:FindFirstChild(tabName .. "Scroller")
	if not mainScroller then return end

	local Paragraph = Instance.new("Frame")
	Paragraph.Name = "ParagraphBars"
	Paragraph.Parent = mainScroller
	Paragraph.AnchorPoint = Vector2.new(0.5, 0.5)
	Paragraph.BackgroundColor3 = bgColor
	Paragraph.BackgroundTransparency = bgTransparency
	Paragraph.BorderSizePixel = 0
	Paragraph.Size = UDim2.new(0, wrapWidth, 0, paragraphSize)
	Paragraph.ZIndex = -1

	local Corner = Instance.new("UICorner")
	Corner.CornerRadius = UDim.new(0, 4)
	Corner.Parent = Paragraph

	CommonUIStroke(Paragraph)

	local Stroke = Paragraph:FindFirstChildOfClass("UIStroke")
	if Stroke then
		Stroke.Transparency = strokeTransparency
		Stroke.Thickness = strokeThickness
		Stroke.Color = strokeColor
	end

	local Glow = Instance.new("ImageLabel")
	Glow.Name = "Glow"
	Glow.Parent = Paragraph
	Glow.BackgroundTransparency = 1
	Glow.AnchorPoint = Vector2.new(0.5, 0.5)
	Glow.Position = UDim2.fromScale(0.5, 0.5)
	Glow.Size = UDim2.new(1, 30, 1, 30)
	Glow.Image = "rbxassetid://1316045217"
	Glow.ImageTransparency = 0.94
	Glow.ImageColor3 = glowColor
	Glow.ZIndex = Paragraph.ZIndex - 1

	local TitleWrap = Instance.new("Frame")
	TitleWrap.Name = "TitleWrap"
	TitleWrap.Parent = Paragraph
	TitleWrap.BackgroundTransparency = 1
	TitleWrap.Position = UDim2.new(0, paddingX, 0, topPad)
	TitleWrap.Size = UDim2.new(1, -paddingX * 2, 0, titleHeight)

	local Accent = Instance.new("Frame")
	Accent.Name = "Accent"
	Accent.Parent = TitleWrap
	Accent.AnchorPoint = Vector2.new(0, 0.5)
	Accent.Position = UDim2.new(0, 0, 0.5, 0)
	Accent.Size = UDim2.new(0, 6, 0, 6)
	Accent.BackgroundColor3 = accentColor
	Accent.BackgroundTransparency = 0.1
	Accent.BorderSizePixel = 0

	local AccentCorner = Instance.new("UICorner")
	AccentCorner.CornerRadius = UDim.new(1, 0)
	AccentCorner.Parent = Accent

	local AccentStroke = Instance.new("UIStroke")
	AccentStroke.Parent = Accent
	AccentStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
	AccentStroke.Thickness = 1
	AccentStroke.Color = accentColor
	AccentStroke.Transparency = 0.55

	local ParagraphTitle = Instance.new("TextLabel")
	ParagraphTitle.Name = "ParagraphTitle"
	ParagraphTitle.Parent = TitleWrap
	ParagraphTitle.BackgroundTransparency = 1
	ParagraphTitle.Position = UDim2.new(0, 12, 0, 0)
	ParagraphTitle.Size = UDim2.new(1, -12, 1, 0)
	ParagraphTitle.FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.SemiBold, Enum.FontStyle.Normal)
	ParagraphTitle.Text = mainTitle or ""
	ParagraphTitle.TextColor3 = Color3.fromRGB(225, 228, 238)
	ParagraphTitle.TextSize = 14
	ParagraphTitle.TextStrokeTransparency = 0.85
	ParagraphTitle.TextWrapped = true
	ParagraphTitle.TextXAlignment = Enum.TextXAlignment.Left

	local Divider = Instance.new("Frame")
	Divider.Name = "Divider"
	Divider.Parent = Paragraph
	Divider.AnchorPoint = Vector2.new(0.5, 0)
	Divider.Position = UDim2.new(0.5, 0, 0, dividerY)
	Divider.Size = UDim2.new(1, -(paddingX * 2 - 2), 0, 2)
	Divider.BackgroundColor3 = glowColor
	Divider.BackgroundTransparency = 0.88
	Divider.BorderSizePixel = 0

	local DividerCorner = Instance.new("UICorner")
	DividerCorner.CornerRadius = UDim.new(1, 0)
	DividerCorner.Parent = Divider

	local ParagraphContent
	if showContent then
		ParagraphContent = Instance.new("TextLabel")
		ParagraphContent.Name = "ParagraphContent"
		ParagraphContent.Parent = Paragraph
		ParagraphContent.BackgroundTransparency = 1
		ParagraphContent.Position = UDim2.new(0, paddingX, 0, contentY)
		ParagraphContent.Size = UDim2.new(1, -paddingX * 2, 0, contentHeight)
		ParagraphContent.FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.Medium, Enum.FontStyle.Normal)

		local s1 = stronger and math.max(0, strokeTransparency - 0.28) or math.max(0, strokeTransparency - 0.18)
		local g1 = stronger and 0.76 or 0.82

		if Stroke then TweenService:Create(Stroke, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = s1}):Play() end
		TweenService:Create(Glow, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageTransparency = g1}):Play()

		local function setCountText(p, g)
			count.Text = tostring(p) .. "/" .. tostring(g)

		local function animateCount(fromValue, toValue, duration)
			countAnimId = countAnimId + 1
			local myId = countAnimId
			local start = os.clock()
			duration = math.max(0.05, duration or 0.25)

			local conn
			conn = RunService.RenderStepped:Connect(function()
				if myId ~= countAnimId then
					if conn then conn:Disconnect() end
					return
				end
				if not count or not count.Parent then
					if conn then conn:Disconnect() end
					return
				end
				local t = (os.clock() - start) / duration
				if t >= 1 then t = 1 end
				local v = math.floor(fromValue + (toValue - fromValue) * t + 0.5)
				setCountText(v, goal)
				if t == 1 then
					if conn then conn:Disconnect() end
				end
			end)
		end

		local function applyVisual(p, g, tweenTime, strongerPulse)
			local ratio = math.clamp(p / g, 0, 1)
			local t = tweenTime or 0.35
			local complete = (p >= g)

			local activeColor = complete and barAccentComplete or barAccent

			if Segments then
				if SegStroke then
					SegStroke.Color = activeColor
					TweenService:Create(SegStroke, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.6}):Play()
					task.delay(0.12, function()
						if SegStroke and SegStroke.Parent then
							TweenService:Create(SegStroke, TweenInfo.new(0.22, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.75}):Play()
						end
					end)
				end

				local lit = complete and segmentsCount or math.floor(ratio * segmentsCount + 0.0001)
				for i = 1, segmentsCount do
					local seg = Segments[i]
					if seg then
						seg.BackgroundColor3 = activeColor
						local target = (i <= lit) and 0.2 or 0.85
						TweenService:Create(seg, TweenInfo.new(t * 0.6, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
							BackgroundTransparency = target
						}):Play()
					end
				end
			elseif BarFill then
				if BarStroke then
					BarStroke.Color = activeColor
					TweenService:Create(BarStroke, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.6}):Play()
					task.delay(0.12, function()
						if BarStroke and BarStroke.Parent then
							TweenService:Create(BarStroke, TweenInfo.new(0.22, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.75}):Play()
						end
					end)
				end

				BarFill.BackgroundColor3 = activeColor
				TweenService:Create(BarFill, TweenInfo.new(t, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
					Size = UDim2.new(complete and 1 or ratio, 0, 1, 0)
				}):Play()
			end

			if rowSep and complete then
				rowSep.BackgroundTransparency = math.max(0.85, rowDividerTransparency - 0.06)
				task.delay(0.2, function()
					if rowSep and rowSep.Parent then
						rowSep.BackgroundTransparency = rowDividerTransparency
					end
				end)
			end

			pulseCore(strongerPulse or complete)
		end

		applyVisual(progress, goal, 0, false)

					if myUpdate ~= updateId then return end
					applyVisual(newProgress, self.Goal, t, self.Emphasis)
				end)
			end,
			SetGoal = function(self, newGoal, tweenTime)
				updateId = updateId + 1
				local myUpdate = updateId

				newGoal = math.max(1, tonumber(newGoal) or 1)
				self.Goal = newGoal
				self.Progress = math.clamp(self.Progress, 0, self.Goal)

				local t = tweenTime or 0.35
				setCountText(self.Progress, self.Goal)

				task.delay(microDelay, function()
					if myUpdate ~= updateId then return end
					applyVisual(self.Progress, self.Goal, t, self.Emphasis)
				end)
			end,
			SetLabel = function(self, newLabel)
				self.Label.Text = tostring(newLabel or "")
				pulseCore(false)
			end,
			SetStyle = function(self, newStyle)
				self.Style = tostring(newStyle or self.Style)
			end
		}

		refreshRowDividers()
		return rowObj
	end

		end)

	local obj = {}

	function obj:AddBar(def)
		local r = addBar(def)
		return r
	end

	function obj:RemoveBar(keyOrIndex)
		local rowObj
		if typeof(keyOrIndex) == "number" then
			rowObj = rows[keyOrIndex]
		else
			rowObj = rowsByKey[tostring(keyOrIndex)]
		end
		if not rowObj then return end

		for k, v in pairs(rowsByKey) do
			if v == rowObj then
				rowsByKey[k] = nil
			end
		end

		for i = #rows, 1, -1 do
			if rows[i] == rowObj then
				table.remove(rows, i)
				break
			end
		end

		if rowObj.Row then rowObj.Row:Destroy() end
		rebuildSizeIfAuto()
		refreshRowDividers()
		pulse()
	end

	function obj:GetBar(keyOrIndex)
		if typeof(keyOrIndex) == "number" then
			return rows[keyOrIndex]
		end
		return rowsByKey[tostring(keyOrIndex)]
	end

	function obj:SetProgress(keyOrIndex, newProgress, tweenTime)
		local b = obj:GetBar(keyOrIndex)
		if b then b:SetProgress(newProgress, tweenTime) end
	end

	function obj:SetGoal(keyOrIndex, newGoal, tweenTime)
		local b = obj:GetBar(keyOrIndex)
		if b then b:SetGoal(newGoal, tweenTime) end
	end

	function obj:SetTitle(newTitle)
		ParagraphTitle.Text = newTitle or ""
		pulse()
	end

	function obj:SetContent(newContent)
		if ParagraphContent then
			ParagraphContent.Text = newContent or ""
			pulse()
		end
	end

	function obj:ListBars()
		return rows
	end

	return obj
end

local function removeFromActive(n)
	local idx = table.find(activeNotifications, n)
	if idx then
		table.remove(activeNotifications, idx)
	end
end

local function dismissNotification(Notification)
	if not Notification or not Notification.Parent then
		return
	end
	if Notification:GetAttribute("Dismissing") then
		return
	end
	Notification:SetAttribute("Dismissing", true)

	local Scale = Notification:FindFirstChildOfClass("UIScale")
	local Stroke = Notification:FindFirstChildOfClass("UIStroke")
	local Shadow = Notification:FindFirstChild("Shadow")
	local Glow = Notification:FindFirstChild("Glow")
	local Top = Notification:FindFirstChild("Top")
	local Accent = Notification:FindFirstChild("Accent")
	local Title = Top and Top:FindFirstChild("Title")
	local Content = Notification:FindFirstChild("Content")
	local Bar = Notification:FindFirstChild("Bar")
	local ButtonsHolder = Notification:FindFirstChild("ButtonsHolder")
	local Dim = Notification:FindFirstChild("Dim")

	if ButtonsHolder then
		for _, v in ipairs(ButtonsHolder:GetDescendants()) do
			if v:IsA("TextButton") then
				v.Active = false
				v.AutoButtonColor = false
			end
		end
	end

	if Scale then tween(Scale, OUT_INFO, {Scale = 0.96}) end
	if Stroke then tween(Stroke, OUT_INFO, {Transparency = 1}) end
	if Shadow then tween(Shadow, OUT_INFO, {ImageTransparency = 1}) end
	if Glow then tween(Glow, OUT_INFO, {ImageTransparency = 1}) end
	if Title then tween(Title, OUT_INFO, {TextTransparency = 1}) end
	if Content then tween(Content, OUT_INFO, {TextTransparency = 1}) end
	if Top then tween(Top, OUT_INFO, {BackgroundTransparency = 1}) end
	if Accent then tween(Accent, OUT_INFO, {BackgroundTransparency = 1}) end
	if Bar then tween(Bar, OUT_INFO, {BackgroundTransparency = 1}) end
	if Dim then tween(Dim, OUT_INFO, {BackgroundTransparency = 1}) end

	local p = Notification.Position
	local outTween = tween(Notification, OUT_INFO, {
		Position = UDim2.new(p.X.Scale, p.X.Offset, p.Y.Scale, p.Y.Offset - 10),
		BackgroundTransparency = 1
	})

	outTween.Completed:Once(function()
		removeFromActive(Notification)
		if Notification and Notification.Parent then
			Notification:Destroy()
		end
	end)
end

local function enforceMaxActive()
	while #activeNotifications >= MAX_ACTIVE do
		local oldest = activeNotifications[1]
		if oldest then
			dismissNotification(oldest)
		else
			break
		end
		task.wait()
	end
end

local function animateIn(Notification, refs)
	Notification.Position = Notification.Position - UDim2.new(0, 0, 0, 12)
	tween(Notification, IN_POS_INFO, {
		Position = Notification.Position + UDim2.new(0, 0, 0, 12),
		BackgroundTransparency = 0.02
	})
	tween(refs.Scale, IN_POP_INFO, {Scale = 1})
	tween(refs.Top, IN_FADE_INFO, {BackgroundTransparency = 0})
	tween(refs.Stroke, IN_FADE_INFO, {Transparency = 0.45})
	tween(refs.Shadow, IN_FADE_INFO, {ImageTransparency = 0.65})
	tween(refs.Glow, IN_FADE_INFO, {ImageTransparency = 0.82})
	tween(refs.Title, IN_FADE_INFO, {TextTransparency = 0})
	tween(refs.Content, IN_FADE_INFO, {TextTransparency = 0})
	tween(refs.Accent, IN_FADE_INFO, {BackgroundTransparency = 0.15})
	tween(refs.Bar, IN_FADE_INFO, {BackgroundTransparency = 0.0})
	tween(refs.Dim, IN_FADE_INFO, {BackgroundTransparency = 0.0})
end

local function showNormalNotification(params)
	enforceMaxActive()

	local titleText = params.TitleText or "Notification"
	local contentText = params.ContentText or ""
	local duration = params.Duration or 3

	local Notification = Instance.new("Frame")
	local refs = buildBase(Notification, titleText, contentText, 66)

	table.insert(activeNotifications, Notification)

	refs.Click.Activated:Connect(function()
		dismissNotification(Notification)
	end)

	animateIn(Notification, refs)

	local barTween = tween(refs.Bar, TweenInfo.new(duration, Enum.EasingStyle.Linear), {Size = UDim2.new(0, 0, 1, 0)})

	task.delay(duration, function()
		if Notification and Notification.Parent and not Notification:GetAttribute("Dismissing") then
			dismissNotification(Notification)
		end
	end)

	Notification.Destroying:Connect(function()
		if barTween then
			pcall(function() barTween:Cancel() end)
		end
	end)
end

local function showInteractableNotification(params)
	enforceMaxActive()

	local titleText = params.TitleText or "Choose"
	local contentText = params.ContentText or ""
	local duration = params.Duration
	local choices = params.Choices
	local onTimeout = params.OnTimeout

	if typeof(choices) ~= "table" or #choices == 0 then
		choices = {
			{Text = "OK", Callback = params.OnAccept or params.Callback}
		}
	end

	local btnCount = math.clamp(#choices, 1, 3)
	local height = 66 + 38

	local Notification = Instance.new("Frame")
	local refs = buildBase(Notification, titleText, contentText, height)

	local ButtonsHolder = Instance.new("Frame")
	ButtonsHolder.Name = "ButtonsHolder"
	ButtonsHolder.Parent = Notification
	ButtonsHolder.BackgroundTransparency = 1
	ButtonsHolder.Position = UDim2.new(0, 12, 0, 66)
	ButtonsHolder.Size = UDim2.new(1, -24, 0, 30)
	ButtonsHolder.ZIndex = Notification.ZIndex + 20

	local layout = Instance.new("UIListLayout")
	layout.FillDirection = Enum.FillDirection.Horizontal
	layout.HorizontalAlignment = Enum.HorizontalAlignment.Right
	layout.VerticalAlignment = Enum.VerticalAlignment.Center
	layout.Padding = UDim.new(0, 8)
	layout.SortOrder = Enum.SortOrder.LayoutOrder
	layout.Parent = ButtonsHolder

	local function makeBtn(text, order)
		local b = Instance.new("TextButton")
		b.Name = "Choice"
		b.Parent = ButtonsHolder
		b.LayoutOrder = order
		b.AutoButtonColor = false
		b.BackgroundColor3 = BTN_BG
		b.BackgroundTransparency = 1
		b.BorderSizePixel = 0
		b.Size = UDim2.new(0, math.floor((ButtonsHolder.AbsoluteSize.X - (btnCount - 1) * 8) / btnCount), 0, 28)
		b.Text = text
		b.TextColor3 = BTN_TEXT
		b.TextSize = 14
		b.FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.Bold, Enum.FontStyle.Normal)
		b.ZIndex = Notification.ZIndex + 30

		local c = Instance.new("UICorner")
		c.CornerRadius = UDim.new(0, 4)
		c.Parent = b

		local s = Instance.new("UIStroke")
		s.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
		s.Color = STROKE_COLOR
		s.Thickness = 1
		s.Transparency = 1
		s.Parent = b

		tween(b, IN_FADE_INFO, {BackgroundTransparency = 0.0})
		tween(s, IN_FADE_INFO, {Transparency = 0.7})

		b.MouseEnter:Connect(function()
			if Notification:GetAttribute("Dismissing") then return end
			tween(b, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundColor3 = BTN_BG_HOVER})
		end)

		b.MouseLeave:Connect(function()
			if Notification:GetAttribute("Dismissing") then return end
			tween(b, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundColor3 = BTN_BG})
		end)

		return b
	end

	local chosen = false
	local function pick(i)
		if chosen then return end
		chosen = true

		local choice = choices[i]
		if choice and typeof(choice.Callback) == "function" then
			task.spawn(function()
				pcall(choice.Callback)
			end)
		end

		dismissNotification(Notification)
	end

	for i = 1, btnCount do
		local c = choices[i]
		local text = (c and c.Text) or ("Choice " .. i)
		local btn = makeBtn(text, i)
		btn.Activated:Connect(function()
			pick(i)
		end)
	end

	table.insert(activeNotifications, Notification)

	refs.Click.Active = false
	refs.Click.Visible = false

	animateIn(Notification, refs)

	if duration and duration > 0 then
		local barTween = tween(refs.Bar, TweenInfo.new(duration, Enum.EasingStyle.Linear), {Size = UDim2.new(0, 0, 1, 0)})

		task.delay(duration, function()
			if Notification and Notification.Parent and not Notification:GetAttribute("Dismissing") then
				if typeof(onTimeout) == "function" then
					task.spawn(function()
						pcall(onTimeout)
					end)
				end
				dismissNotification(Notification)
			end
		end)

		Notification.Destroying:Connect(function()
			if barTween then
				pcall(function() barTween:Cancel() end)
			end
		end)
	else
		refs.Bar.BackgroundTransparency = 1
	end
end

local function showNotification(params)
	local kind = params.Kind
	if kind == "Interactable" then
		showInteractableNotification(params)
	else
		showNormalNotification(params)
	end
end

local function processQueue()
	isProcessingQueue = true
	task.spawn(function()
		while #notificationQueue > 0 do
			local params = table.remove(notificationQueue, 1)
			showNotification(params)
			task.wait(QUEUE_GAP)
		end
		isProcessingQueue = false
	end)
end

function PrestineLib:AddNotification(params)
	params = params or {}
	params.Kind = params.Kind or "Normal"
	table.insert(notificationQueue, params)
	if not isProcessingQueue then
		processQueue()
	end
end

function PrestineLib:AddInteractableNotif(params)
	params = params or {}
	params.Kind = "Interactable"
	table.insert(notificationQueue, params)
	if not isProcessingQueue then
		processQueue()
	end
end

			IconGlowImage.ImageTransparency = tGlow
			AmbientGlowImage.ImageTransparency = (hovering and 0.82 or 0.88) + (math.sin(breathe * 1.7) * 0.02)
			local a = localAccent[i]
			local c = localAccent[n]
			local tt = 0
			while tt < 1 and ButtonContainer.Parent do
				local dtt = RS.RenderStepped:Wait()
				breathe = breathe + dtt
				tt = tt + (dtt * 0.7)

				IconContainer.BackgroundColor3 = col
				iconStroke.Color = col

				else
					outerStroke.Transparency = 0.58 + (math.sin(breathe * 2) * 0.02)
				end
			end
			i = n
			n = n % #localAccent + 1
		end
	end)

end

	local function sparkBurst()
		local burstCount = 7
		for i = 1, burstCount do
			local dot = Instance.new("Frame")
			dot.Parent = Switch
			dot.AnchorPoint = Vector2.new(0.5, 0.5)
			dot.Position = Knob.Position
			dot.Size = UDim2.new(0, 4, 0, 4)
			dot.BackgroundColor3 = Color3.fromRGB(110, 160, 235)
			dot.BackgroundTransparency = 0.15
			dot.BorderSizePixel = 0
			dot.ZIndex = Knob.ZIndex + 2

			local dc = Instance.new("UICorner")
			dc.CornerRadius = UDim.new(1, 0)
			dc.Parent = dot

			local angle = (math.pi * 2) * (i / burstCount)
			local radius = 8 + math.random(0, 6)
			local ox = math.cos(angle) * radius
			local oy = math.sin(angle) * radius

			TweenService:Create(dot, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
				Position = dot.Position + UDim2.new(0, ox, 0, oy),
				BackgroundTransparency = 1,
				Size = UDim2.new(0, 8, 0, 8)
			}):Play()

			task.delay(0.19, function()
				if dot then dot:Destroy() end
			end)
		end
	end

		local knobPos = state and UDim2.new(1, -11, 0.5, 0) or UDim2.new(0, 11, 0.5, 0)
		local knobCol = state and Color3.fromRGB(110, 160, 235) or Color3.fromRGB(65, 70, 85)
		local switchCol = state and Color3.fromRGB(10, 30, 45) or Color3.fromRGB(20, 22, 32)

		if instant then
			Knob.Position = knobPos
			Knob.BackgroundColor3 = knobCol
			Switch.BackgroundColor3 = switchCol

			GlowStroke.Transparency = state and 0.65 or 1
			Glow.BackgroundTransparency = state and 0.9 or 1

			SwitchStroke.Transparency = state and 0.25 or 0.55
			SwitchStroke.Color = state and Color3.fromRGB(90, 140, 220) or Color3.fromRGB(30, 35, 50)

			RowGlowStroke.Transparency = state and 0.78 or 1
			RowGlow.BackgroundTransparency = state and 0.94 or 1
			RowStroke.Transparency = state and 0.35 or 0.55
			RowStroke.Color = state and Color3.fromRGB(90, 140, 220) or Color3.fromRGB(30, 35, 50)

			setOnPill(state, true)
			return
		end

		TweenService:Create(Knob, TweenInfo.new(0.18, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {
			Position = knobPos,
			BackgroundColor3 = knobCol
		}):Play()

			BackgroundColor3 = switchCol
		}):Play()

			Transparency = state and 0.25 or 0.55
		}):Play()

			Transparency = state and 0.65 or 1
		}):Play()

		TweenService:Create(Glow, TweenInfo.new(0.22, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
			BackgroundTransparency = state and 0.9 or 1
		}):Play()

		TweenService:Create(RowStroke, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
			Transparency = state and 0.35 or 0.55,

			BackgroundTransparency = state and 0.94 or 1
		}):Play()

		setOnPill(state, false)

		if state then
			pulse()
			sparkBurst()
		else
			sparkBurst()
		end
	end

	end)

	local function fmt(sec)
		sec = math.max(0, math.floor(sec + 0.5))
		if sec >= 604800 then
			return string.format("%dw %dd", math.floor(sec / 604800), math.floor((sec % 604800) / 86400))
		elseif sec >= 86400 then
			return string.format("%dd %dh", math.floor(sec / 86400), math.floor((sec % 86400) / 3600))
		elseif sec >= 3600 then
			return string.format("%dh %dm", math.floor(sec / 3600), math.floor((sec % 3600) / 60))
		elseif sec >= 60 then
			return string.format("%dm %ds", math.floor(sec / 60), sec % 60)
		else
			return string.format("%ds", sec)
		end
	end

	local timerToken = 0
	local endAt = nil
	local countdownConn = nil

	local function stopCountdown()
		endAt = nil
		if countdownConn then countdownConn:Disconnect() countdownConn = nil end
		Countdown.Text = (mode == "autoon") and "Auto ON" or "Auto OFF"
		Countdown.TextColor3 = Color3.fromRGB(110, 110, 110)
	end

	local function startCountdown()
		stopCountdown()
		endAt = os.clock() + seconds()
		Countdown.TextColor3 = Color3.fromRGB(150, 150, 150)
		countdownConn = RunService.Heartbeat:Connect(function()
			if not Toggle or not Toggle.Parent then
				stopCountdown()
				return
			end
			if not endAt then return end
			local left = endAt - os.clock()
			if left <= 0 then
				Countdown.Text = "Now"
				stopCountdown()
				return
			end
			Countdown.Text = fmt(left)
		end)
	end

	local function shouldArmTimer()
		if mode == "autoon" then
			return toggleState == false
		end
		return toggleState == true
	end

	local function applyVisual(instant)
		local knobPos = toggleState and UDim2.new(1, -11, 0.5, 0) or UDim2.new(0, 11, 0.5, 0)
		local knobCol = toggleState and Color3.fromRGB(110, 160, 235) or Color3.fromRGB(65, 70, 85)
		local swCol = toggleState and Color3.fromRGB(10, 30, 45) or Color3.fromRGB(20, 22, 32)
		local st = toggleState and 0.25 or 0.55
		if instant then
			Knob.Position = knobPos
			Knob.BackgroundColor3 = knobCol
			Switch.BackgroundColor3 = swCol
			SwitchStroke.Transparency = st
		else
			TweenService:Create(Knob, TweenInfo.new(0.18, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), { Position = knobPos, BackgroundColor3 = knobCol }):Play()
			TweenService:Create(Switch, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { BackgroundColor3 = swCol }):Play()
			TweenService:Create(SwitchStroke, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { Transparency = st }):Play()
		end
	end

	local function scheduleTimer()
		timerToken = timerToken + 1
		local myToken = timerToken
		startCountdown()
		local s = seconds()
		task.delay(s, function()
			if myToken ~= timerToken then return end
			if not Toggle or not Toggle.Parent then return end
			if mode == "autoon" then
				if not toggleState then toggleState = true end
			else
				if toggleState then toggleState = false end
			end
			persist()
			applyVisual(false)
			callback(toggleState)
			stopCountdown()
		end)
	end

	local unitOrder = { "s", "m", "h", "d", "w" }
	local function cycleUnit()
		for i, u in ipairs(unitOrder) do
			if u == unitVal then
				unitVal = unitOrder[(i % #unitOrder) + 1]
				break
			end
		end
		UnitBtn.Text = unitVal:upper()
		persist()
		timerToken = timerToken + 1
		if shouldArmTimer() then scheduleTimer() else stopCountdown() end
	end

	local function sanitizeDurationText()
		local t = DurationBox.Text or ""
		t = t:gsub("%D", "")
		if t == "" then t = "1" end
		local n = math.clamp(tonumber(t) or 1, 1, 999999)
		durationVal = n
		DurationBox.Text = tostring(n)
		persist()
		timerToken = timerToken + 1
		if shouldArmTimer() then scheduleTimer() else stopCountdown() end
	end

	DurationBox:GetPropertyChangedSignal("Text"):Connect(function()
		local t = DurationBox.Text or ""
		local cleaned = t:gsub("%D", "")
		if cleaned ~= t then
			local pos = DurationBox.CursorPosition
			DurationBox.Text = cleaned
			DurationBox.CursorPosition = math.max(1, math.min(pos - 1, #cleaned + 1))
		end
	end)

	DurationBox.FocusLost:Connect(function()
		sanitizeDurationText()
	end)

	UnitBtn.MouseButton1Click:Connect(function()
		cycleUnit()
	end)

	Hitbox.MouseButton1Down:Connect(function()
		TweenService:Create(KnobScale, TweenInfo.new(0.08, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { Scale = 0.94 }):Play()
	end)

	Hitbox.MouseButton1Up:Connect(function()
		TweenService:Create(KnobScale, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { Scale = 1 }):Play()
	end)

	Hitbox.MouseButton1Click:Connect(function()
		toggleState = not toggleState
		timerToken = timerToken + 1
		applyVisual(false)
		persist()
		callback(toggleState)
		if shouldArmTimer() then scheduleTimer() else stopCountdown() end
	end)

	applyVisual(true)
	persist()
	task.spawn(function() callback(toggleState) end)

	if shouldArmTimer() then
		scheduleTimer()
	else
		stopCountdown()
	end

	Toggle.Destroying:Connect(function()
		stopCountdown()
		timerToken = timerToken + 1
	end)

	return {
		Set = function(_, v)
			toggleState = (v == true)
			timerToken = timerToken + 1
			applyVisual(false)
			persist()
			callback(toggleState)
			if shouldArmTimer() then scheduleTimer() else stopCountdown() end
		end,
		Get = function()
			return toggleState, durationVal, unitVal
		end,
		SetTimer = function(_, dur, unit)
			durationVal = math.clamp(tonumber(dur) or durationVal, 1, 999999)
			unit = (unit or unitVal):lower()
			if unit == "s" or unit == "m" or unit == "h" or unit == "d" or unit == "w" then
				unitVal = unit
			end
			DurationBox.Text = tostring(durationVal)
			UnitBtn.Text = unitVal:upper()
			persist()
			timerToken = timerToken + 1
			if shouldArmTimer() then scheduleTimer() else stopCountdown() end
		end
	}
end

	local buttons = {}
	local open = false

	if multiple then
		if typeof(defaultChoice) == "table" then
			for _, c in ipairs(defaultChoice) do
				selectedSet[c] = true
			end
		end
	else
		selectedSingle = defaultChoice
	end

	local function buildSelectedArray()
		local out = {}
		for _, e in ipairs(buttons) do
			if selectedSet[e.choice] then
				out[#out + 1] = e.choice
			end
		end
		return out
	end

	local function persist(value)
		if _G.isSavingGV then
			PrestineLib:SetValue(mainTitle, value)
			PrestineLib:Save()
		end
	end

	local function updateSelectedText()
		if multiple then
			local arr = buildSelectedArray()
			if #arr == 0 then
				SelectedText.Text = "None"
			elseif #arr <= 2 then
				SelectedText.Text = table.concat(arr, ", ")
			else
				SelectedText.Text = tostring(#arr) .. " selected"
			end
			callback(arr)
			persist(arr)
		else
			local v = selectedSingle
			SelectedText.Text = (v and tostring(v) ~= "" and v) or "None"
			callback(v)
			persist({ SelectedText.Text })
		end
	end

	local function applyChoiceVisual(entry, isSel)
		entry.btn.BackgroundTransparency = isSel and 0.08 or 0.22
		entry.stroke.Transparency = isSel and 0.25 or 0.55
		entry.stroke.Color = isSel and Color3.fromRGB(100, 150, 230) or Color3.fromRGB(30, 35, 50)
		entry.label.TextColor3 = isSel and Color3.fromRGB(220, 240, 255) or Color3.fromRGB(180, 180, 180)
	end

		local btn = Instance.new("TextButton")
		btn.Parent = List

		btn.BackgroundColor3 = Color3.fromRGB(14, 16, 26)
		btn.BackgroundTransparency = 0.22
		btn.BorderSizePixel = 0
		btn.AutoButtonColor = false
		btn.Text = ""
		btn.LayoutOrder = i
		btn.ZIndex = 31

		local bc = Instance.new("UICorner", btn)
		bc.CornerRadius = UDim.new(0, 4)

		local stroke = Instance.new("UIStroke", btn)
		stroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
		stroke.Thickness = 1.4
		stroke.Color = Color3.fromRGB(30, 35, 50)
		stroke.Transparency = 0.55

		local label = Instance.new("TextLabel")
		label.Parent = btn
		label.BackgroundTransparency = 1
		label.Position = UDim2.new(0, 12, 0, 0)
		label.Size = UDim2.new(1, -24, 1, 0)
		label.FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.SemiBold, Enum.FontStyle.Normal)
		label.Text = tostring(choice)
		label.TextColor3 = Color3.fromRGB(180, 180, 180)
		label.TextSize = 14
		label.TextXAlignment = Enum.TextXAlignment.Left
		label.ZIndex = 32

		local scale = Instance.new("UIScale", btn)
		scale.Scale = 1

		local entry = {btn = btn, stroke = stroke, label = label, choice = choice, scale = scale}
		buttons[#buttons + 1] = entry

		local isSel = multiple and selectedSet[choice] == true or (not multiple and selectedSingle == choice)
		applyChoiceVisual(entry, isSel)

		btn.MouseEnter:Connect(function()
			tween(scale, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Scale = 1.02})
			tween(stroke, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = math.max(stroke.Transparency - 0.12, 0.18)})

		btn.MouseLeave:Connect(function()
			tween(scale, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Scale = 1})
			local sel = multiple and selectedSet[choice] == true or (not multiple and selectedSingle == choice)
			tween(stroke, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = sel and 0.25 or 0.55})
		end)

		btn.MouseButton1Click:Connect(function()
			if not multiple then
				selectedSingle = choice
				for _, e in ipairs(buttons) do applyChoiceVisual(e, false) end
				applyChoiceVisual(entry, true)
				updateSelectedText()
				open = false
				setChevron(false)
				tween(HeaderGlow, TweenInfo.new(0.16, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageTransparency = 1})

				local t = tween(DropdownContent, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 1})
				tween(ContentScale, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Scale = 0.98})
				tween(ContentStroke, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 1})
				tween(ContentGlow, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageTransparency = 1})

					if DropdownContent then DropdownContent.Visible = false end

			updateSelectedText()
		end)

	updateSelectedText()

	local function updateSearch()
		local q = string.lower(SearchBox.Text or "")
		for _, entry in ipairs(buttons) do
			if q == "" then
				entry.btn.Visible = true
			else
				entry.btn.Visible = string.find(string.lower(tostring(entry.choice)), q, 1, true) ~= nil
			end
		end
	end
	SearchBox:GetPropertyChangedSignal("Text"):Connect(updateSearch)

	local closingGuard = false
	local function openDropdown()
		if open or closingGuard then return end
		open = true
		Dropdown.ZIndex = 10

		setChevron(true)
		tween(RootScale, TweenInfo.new(0.10, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Scale = 0.985})
		task.delay(0.10, function()
			if Dropdown and Dropdown.Parent then
				tween(RootScale, TweenInfo.new(0.18, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Scale = 1})
			end
		end)

		tween(HeaderGlow, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageTransparency = 0.93})
		if RootStroke then tween(RootStroke, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.35}) end
		tween(PillStroke, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.55})

		DropdownContent.Visible = true
		DropdownContent.BackgroundTransparency = 1
		ContentScale.Scale = 0.96
		DropdownContent.Position = UDim2.new(0.5, 0, 1, 2)

		tween(DropdownContent, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0.04})
		tween(ContentScale, TweenInfo.new(0.22, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Scale = 1})
		tween(DropdownContent, TweenInfo.new(0.22, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Position = UDim2.new(0.5, 0, 1, 6)})
		tween(ContentStroke, TweenInfo.new(0.20, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.55})
		tween(ContentGlow, TweenInfo.new(0.22, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageTransparency = 0.92})
	end

	local function closeDropdown()
		if (not open) or closingGuard then return end
		open = false
		closingGuard = true

		setChevron(false)
		tween(HeaderGlow, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageTransparency = 1})
		if RootStroke then tween(RootStroke, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.55}) end
		tween(PillStroke, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.75})

		tween(ContentScale, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Scale = 0.98})
		tween(ContentStroke, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 1})
		tween(ContentGlow, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageTransparency = 1})
		local t = tween(DropdownContent, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
			BackgroundTransparency = 1,
			Position = UDim2.new(0.5, 0, 1, 2)
		})
		t.Completed:Once(function()
			if DropdownContent and DropdownContent.Parent then
				DropdownContent.Visible = false
				DropdownContent.Position = UDim2.new(0.5, 0, 1, 6)
			end
			Dropdown.ZIndex = 0
			closingGuard = false
		end)
	end

	MainButton.MouseButton1Click:Connect(function()
		if open then closeDropdown() else openDropdown() end
	end)

	local outsideConn
	outsideConn = UIS.InputBegan:Connect(function(input, gp)
		if gp then return end
		if not open then return end
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			local pos = input.Position
			local inHeader = (pos.X >= Dropdown.AbsolutePosition.X and pos.X <= Dropdown.AbsolutePosition.X + Dropdown.AbsoluteSize.X
				and pos.Y >= Dropdown.AbsolutePosition.Y and pos.Y <= Dropdown.AbsolutePosition.Y + Dropdown.AbsoluteSize.Y)
			local inContent = DropdownContent.Visible and
				(pos.X >= DropdownContent.AbsolutePosition.X and pos.X <= DropdownContent.AbsolutePosition.X + DropdownContent.AbsoluteSize.X
				and pos.Y >= DropdownContent.AbsolutePosition.Y and pos.Y <= DropdownContent.AbsolutePosition.Y + DropdownContent.AbsoluteSize.Y)
			if not inHeader and not inContent then
				closeDropdown()
			end
		end
	end)

	Dropdown.Destroying:Connect(function()
		if outsideConn then outsideConn:Disconnect() end
	end)

	local dropdownObject = {
		UpdateDropdown = function(self, newList)
			for _, entry in ipairs(buttons) do entry.btn:Destroy() end
			table.clear(buttons)

			choiceList = newList or {}

			local exists = {}
			for _, c in ipairs(choiceList) do exists[c] = true end

			if multiple then
				for c in pairs(selectedSet) do
					if not exists[c] then selectedSet[c] = nil end
				end
			else
				if selectedSingle and not exists[selectedSingle] then selectedSingle = nil end
			end

			for i, c in ipairs(choiceList) do createChoice(i, c) end
			updateSearch()
			updateSelectedText()
		end,
		Close = function()
			closeDropdown()
		end
	}

	return dropdownObject
end

function PrestineLib:AddSlider(params)
	local tabName = params.Tab
	local sliderTitle = params.SliderTitle or "Slider"
	local minValue = params.Min or 0
	local maxValue = params.Max or 100
	local step = math.max(params.Increment or 1, 1)

	local savedValue = PrestineLib:GetValue(sliderTitle)
	local initialValue = tonumber(savedValue) or params.DefaultValue or minValue
	initialValue = math.clamp(initialValue, minValue, maxValue)

	local targetTab = Main:FindFirstChild(tabName .. "Contents")
	if not targetTab then return end

	local mainScroller = targetTab:FindFirstChild(tabName .. "Scroller")
	if not mainScroller then return end

	local Slider = Instance.new("Frame")
	Slider.Name = removeSpaces(sliderTitle)
	Slider.Parent = mainScroller
	Slider.AnchorPoint = Vector2.new(0.5, 0.5)
	Slider.BackgroundColor3 = Color3.fromRGB(16, 18, 28)
	Slider.Size = UDim2.new(0, 325, 0, 60)

	local UICornerS = Instance.new("UICorner")
	UICornerS.CornerRadius = UDim.new(0, 4)
	UICornerS.Parent = Slider

	CommonUIStroke(Slider)

	local SliderLabel = Instance.new("TextLabel")
	SliderLabel.Parent = Slider
	SliderLabel.Name = "SliderTitle"
	SliderLabel.AnchorPoint = Vector2.new(0.5, 0.5)
	SliderLabel.Position = UDim2.new(0.18, 0, 0.3, 0)
	SliderLabel.Size = UDim2.new(0, 100, 0, 20)
	SliderLabel.Text = sliderTitle
	SliderLabel.BackgroundTransparency = 1
	SliderLabel.TextColor3 = Color3.fromRGB(180, 180, 180)
	SliderLabel.FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.SemiBold, Enum.FontStyle.Normal)
	SliderLabel.TextSize = 14
	SliderLabel.TextXAlignment = Enum.TextXAlignment.Left

	local Track = Instance.new("Frame")
	Track.Name = "Track"
	Track.Parent = Slider
	Track.Position = UDim2.new(0, 10, 0, 35)
	Track.Size = UDim2.new(0.9, 0, 0.25, 0)
	Track.BackgroundColor3 = Color3.fromRGB(80, 80, 80)

	local TrackCorner = Instance.new("UICorner")
	TrackCorner.CornerRadius = UDim.new(1, 0)
	TrackCorner.Parent = Track

	local TrailFill = Instance.new("Frame")
	TrailFill.Name = "TrailFill"
	TrailFill.Parent = Track
	TrailFill.ZIndex = 0
	TrailFill.Size = UDim2.new(0, 0, 1, 0)
	TrailFill.BackgroundColor3 = Color3.fromRGB(110, 160, 235)
	TrailFill.BackgroundTransparency = 0.65
	TrailFill.BorderSizePixel = 0

	local TrailCorner = Instance.new("UICorner")
	TrailCorner.CornerRadius = UDim.new(1, 0)
	TrailCorner.Parent = TrailFill

	local TrailStroke = Instance.new("UIStroke")
	TrailStroke.Name = "TrailStroke"
	TrailStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
	TrailStroke.Thickness = 1
	TrailStroke.Color = Color3.fromRGB(110, 160, 235)
	TrailStroke.Transparency = 0.75
	TrailStroke.Parent = TrailFill

	local Fill = Instance.new("Frame")
	Fill.Name = "Fill"
	Fill.Parent = Track
	Fill.ZIndex = 1
	Fill.Size = UDim2.new(0, 0, 1, 0)
	Fill.BackgroundColor3 = Color3.fromRGB(100, 150, 230)
	Fill.BorderSizePixel = 0

	local FillCorner = Instance.new("UICorner")
	FillCorner.CornerRadius = UDim.new(1, 0)
	FillCorner.Parent = Fill

	local blues = {
		Color3.fromRGB(90, 140, 220),
		Color3.fromRGB(95, 145, 225),
		Color3.fromRGB(75, 125, 210),
	}

	local currentIndex = 1
	local nextIndex = 2
	local transitionSpeed = 2

	task.spawn(function()
		while Fill and Fill.Parent do
			local t = 0
			while t < 1 and Fill and Fill.Parent do
				t = t + transitionSpeed * RunService.RenderStepped:Wait()
				Fill.BackgroundColor3 = Fill.BackgroundColor3:Lerp(blues[nextIndex], t)
			end
			currentIndex = nextIndex
			nextIndex = nextIndex + 1
			if nextIndex > #blues then nextIndex = 1 end
		end
	end)

	local Knob = Instance.new("TextButton")
	Knob.Name = "Knob"
	Knob.Parent = Track
	Knob.Size = UDim2.new(0, 10, 1.5, 0)
	Knob.AnchorPoint = Vector2.new(0.5, 0.5)
	Knob.Position = UDim2.new(0, 0, 0.5, 0)
	Knob.BackgroundColor3 = Color3.fromRGB(14, 18, 28)
	Knob.Text = ""
	Knob.AutoButtonColor = false

	local KnobCorner = Instance.new("UICorner")
	KnobCorner.CornerRadius = UDim.new(1, 0)
	KnobCorner.Parent = Knob

	local ValueLabel = Instance.new("TextLabel")
	ValueLabel.Name = "ValueLabel"
	ValueLabel.Parent = Slider
	ValueLabel.Position = UDim2.new(0, 250, 0, 5)
	ValueLabel.Size = UDim2.new(0, 60, 0, 20)
	ValueLabel.BackgroundTransparency = 1
	ValueLabel.TextColor3 = Color3.fromRGB(180, 180, 180)
	ValueLabel.FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.SemiBold, Enum.FontStyle.Normal)
	ValueLabel.TextSize = 14
	ValueLabel.Text = tostring(initialValue)

	local dragging = false
	local trailTween

	local function updateTrail(width)
		if trailTween then trailTween:Cancel() end
		trailTween = TweenService:Create(
			TrailFill,
			TweenInfo.new(1.25, Enum.EasingStyle.Bounce, Enum.EasingDirection.Out),
			{Size = UDim2.new(0, width, 1, 0)}
		)
		trailTween:Play()
	end

	local function applyValue(rawValue)
		local steppedValue = math.floor((rawValue - minValue) / step + 0.5) * step + minValue
		steppedValue = math.clamp(steppedValue, minValue, maxValue)
		local percent = (steppedValue - minValue) / (maxValue - minValue)
		local width = percent * Track.AbsoluteSize.X

		Fill.Size = UDim2.new(0, width, 1, 0)
		Knob.Position = UDim2.new(0, width, 0.5, 0)
		updateTrail(width)

		ValueLabel.Text = tostring(steppedValue)
		if _G.isSavingGV then
			PrestineLib:SetValue(sliderTitle, steppedValue)
			PrestineLib:Save()
		end
		if params.Callback then
			params.Callback(steppedValue)
		end
	end

	local function setSliderFromX(x)
		local relativeX = math.clamp(x - Track.AbsolutePosition.X, 0, Track.AbsoluteSize.X)
		local percent = relativeX / Track.AbsoluteSize.X
		local rawValue = minValue + percent * (maxValue - minValue)
		applyValue(rawValue)
	end

	Knob.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			dragging = true
			SliderEffectPS(Track, Fill, Knob, "Down")
		end
	end)

	UIS.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			dragging = false
			SliderEffectPS(Track, Fill, Knob, "Up")
		end
	end)

	UIS.InputChanged:Connect(function(input)
		if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
			setSliderFromX(input.Position.X)
			SliderEffectPS(Track, Fill, Knob, "Drag")
		end
	end)

	Track.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			setSliderFromX(input.Position.X)
			SliderEffectPS(Track, Fill, Knob, "Snap")
		end
	end)

	local sliderObject = {
		SetValue = function(self, value)
			applyValue(value)
		end
	}

	Slider:GetPropertyChangedSignal("AbsoluteSize"):Once(function()
		sliderObject:SetValue(initialValue)
	end)

	return sliderObject
end

function PrestineLib:AddKeybind(params)
	local UIS = game:GetService("UserInputService")

	local tabName = params.Tab
	local mainTitle = params.MainTitle
	local defaultKey = params.DefaultKey or Enum.KeyCode.Unknown
	local callback = params.Callback or function() end
	local isExempted = params.isExempted or false

	local savedKey = PrestineLib:GetValue(mainTitle, defaultKey)
	if typeof(savedKey) == "string" then
		local ok, kc = pcall(function() return Enum.KeyCode[savedKey] end)
		savedKey = ok and kc or defaultKey
	end

		savedKey = defaultKey
	end

	local currentKey = savedKey
	local capturing = false
	local captureConn

	local targetTab = Main:FindFirstChild(tabName .. "Contents")

	Pill.Parent = Root

	Pill.Position = UDim2.new(1, -12, 0.5, 0)
	Pill.Size = UDim2.new(0, 110, 0, 28)
	Pill.BackgroundColor3 = Color3.fromRGB(16, 18, 28)
	Pill.BorderSizePixel = 0
	Pill.Text = ""
	Pill.AutoButtonColor = false

	local PillCorner = Instance.new("UICorner")
	PillCorner.CornerRadius = UDim.new(0, 4)
	PillCorner.Parent = Pill

	PillStroke.Color = Color3.fromRGB(100, 180, 255)
	PillStroke.Transparency = 0.55
	PillStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border

	local KeyTag = Instance.new("TextLabel")
	KeyTag.Parent = Pill
	KeyTag.AnchorPoint = Vector2.new(0, 0.5)
	KeyTag.Position = UDim2.new(0, 8, 0.5, 0)
	KeyTag.Size = UDim2.new(0, 24, 0, 16)

	KeyTag.Text = "KEY"
	KeyTag.TextColor3 = Color3.fromRGB(120, 120, 120)
	KeyTag.TextSize = 12
	KeyTag.TextXAlignment = Enum.TextXAlignment.Left

	local Value = Instance.new("TextLabel")
	Value.Name = "KeyValue"
	Value.Parent = Pill
	Value.AnchorPoint = Vector2.new(1, 0.5)
	Value.Position = UDim2.new(1, -8, 0.5, 0)
	Value.Size = UDim2.new(0, 72, 0, 16)
	Value.BackgroundTransparency = 1
	Value.FontFace = Font.new("rbxasset://fonts/families/GothamSSm.json", Enum.FontWeight.Bold, Enum.FontStyle.Normal)
	Value.TextColor3 = Color3.fromRGB(180, 180, 180)
	Value.TextSize = 14
	Value.TextXAlignment = Enum.TextXAlignment.Right
	Value.TextTruncate = Enum.TextTruncate.AtEnd

	local function keyToText(k)
		if typeof(k) ~= "EnumItem" then return "None" end
		if k == Enum.KeyCode.Unknown then return "None" end
		local n = k.Name
		n = n:gsub("^Left", "L "):gsub("^Right", "R ")
		n = n:gsub("Control", "Ctrl")
		return n
	end

	Value.Text = keyToText(currentKey)

	local function setKey(newKey)
		currentKey = newKey
		Value.Text = keyToText(currentKey)

		if _G.isSavingGV or not isExempted then
			PrestineLib:SetValue(mainTitle, currentKey.Name)
			PrestineLib:Save()
		end
		callback(currentKey)
	end

	local function endCapture(commitPulse)
		capturing = false
		if commitPulse then
			InputEffectPS(Pill, PillStroke, "Commit")
		end
		InputEffectPS(Pill, PillStroke, "Unfocus")
		if captureConn then
			captureConn:Disconnect()
			captureConn = nil
		end
	end

	local function beginCapture()
		if capturing then return end
		capturing = true
		Value.Text = "..."
		InputEffectPS(Pill, PillStroke, "Focus")

		if captureConn then captureConn:Disconnect() end
		captureConn = UIS.InputBegan:Connect(function(input, gpe)
			if gpe then return end

			if input.UserInputType == Enum.UserInputType.Keyboard then
				local kc = input.KeyCode
				if kc == Enum.KeyCode.Escape or kc == Enum.KeyCode.Backspace then
					setKey(Enum.KeyCode.Unknown)
				else
					setKey(kc)
				end
				endCapture(true)
				return
			end

			if input.UserInputType == Enum.UserInputType.MouseButton1 then return end
			if input.UserInputType == Enum.UserInputType.MouseButton2 then return end
			if input.UserInputType == Enum.UserInputType.MouseButton3 then return end
			if input.UserInputType == Enum.UserInputType.MouseWheel then return end

			local mapped = input.KeyCode
			if mapped and mapped ~= Enum.KeyCode.Unknown then
				setKey(mapped)
				endCapture(true)
			end
		end)
	end

	Pill.MouseButton1Click:Connect(function()
		beginCapture()
	end)

	Pill.MouseEnter:Connect(function()
		if not capturing then
			InputEffectPS(Pill, PillStroke, "Focus")
		end
	end)

	Pill.MouseLeave:Connect(function()
		if not capturing then
			InputEffectPS(Pill, PillStroke, "Unfocus")
		end
	end)

	task.spawn(function()
		callback(currentKey)
	end)

	local keybindObject = {
		SetKey = function(self, keycode)
			if typeof(keycode) ~= "EnumItem" then return end
			setKey(keycode)
		end,
		GetKey = function(self)
			return currentKey
		end
	}

	return keybindObject
end

	local function clamp255(n)
		n = tonumber(n) or 0
		return math.clamp(n, 0, 255)
	end

	local function setColor(c, fromUser)
		currentColor = c
		Swatch.BackgroundColor3 = currentColor
		SwatchShadow.ImageColor3 = currentColor

		if fromUser then

			SwatchStroke.Transparency = 0.15
			TweenService:Create(SwatchStroke, TweenInfo.new(0.25, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.55}):Play()

			if _G.isSavingGV or not isExempted then
				PrestineLib:SetValue(mainName, {R = currentColor.R, G = currentColor.G, B = currentColor.B})
				PrestineLib:Save()
			end
			callback(currentColor)
		end
	end

	local function syncBoxesFromColor()
		local r = math.floor(currentColor.R * 255 + 0.5)
		local g = math.floor(currentColor.G * 255 + 0.5)
		local b = math.floor(currentColor.B * 255 + 0.5)
		RBox.Text = tostring(r)
		GBox.Text = tostring(g)
		BBox.Text = tostring(b)
	end

	local function updateFromBoxes(fromUser)
		local r = clamp255(RBox.Text)
		local g = clamp255(GBox.Text)
		local b = clamp255(BBox.Text)
		setColor(Color3.fromRGB(r, g, b), fromUser)
	end

	local function digitsOnly(box)
		local lock = false
		box:GetPropertyChangedSignal("Text"):Connect(function()
			if lock then return end
			lock = true
			local t = box.Text:gsub("%D", "")
			if #t > 3 then t = t:sub(1, 3) end
			box.Text = t
			lock = false

			if box:IsFocused() and #box.Text == 3 then
				if box == RBox then
					GBox:CaptureFocus()
				elseif box == GBox then
					BBox:CaptureFocus()
				end
			end

			if RBox.Text ~= "" and GBox.Text ~= "" and BBox.Text ~= "" then
				updateFromBoxes(false)
				RGBStroke.Transparency = 0.25
				TweenService:Create(RGBStroke, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.7}):Play()

		end)

		end)

			InputEffectPS(RGBBg, RGBStroke, "Commit")
			InputEffectPS(RGBBg, RGBStroke, "Unfocus")
			updateFromBoxes(true)
		end)
	end

	digitsOnly(GBox)
	digitsOnly(BBox)

	local PRESET_COUNT = 24
