local function dissolveOut()
	TweenService:Create(UIScaleSU, TweenInfo.new(0.22, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {Scale = 1.05}):Play()
	local t = TweenService:Create(BackgroundSU, TweenInfo.new(0.22, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {BackgroundTransparency = 1})
	t:Play()
	TweenService:Create(GlowSU, TweenInfo.new(0.22, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {Transparency = 1}):Play()
	t.Completed:Once(function()
		if PrestineStartup then PrestineStartup:Destroy() end
		if InstanceScreen then InstanceScreen:Destroy() end
	end)
end

local function buttonFX(btn, icon, hoverColor)
	local scale = Instance.new("UIScale")
	scale.Parent = btn
	scale.Scale = 1

	btn.MouseEnter:Connect(function()
		TweenService:Create(scale,TweenInfo.new(0.12),{Scale=1.1}):Play()
		TweenService:Create(icon,TweenInfo.new(0.12),{ImageColor3=hoverColor}):Play()
	end)

	btn.MouseLeave:Connect(function()
		TweenService:Create(scale,TweenInfo.new(0.14),{Scale=1}):Play()
		TweenService:Create(icon,TweenInfo.new(0.14),{ImageColor3=Color3.fromRGB(180,180,180)}):Play()
	end)

	btn.MouseButton1Down:Connect(function()
		TweenService:Create(scale,TweenInfo.new(0.08),{Scale=0.9}):Play()
	end)

	btn.MouseButton1Up:Connect(function()
		TweenService:Create(scale,TweenInfo.new(0.12,Enum.EasingStyle.Back),{Scale=1}):Play()
	end)
end

buttonFX(ExitButton,ExitIcon,Color3.fromRGB(255,90,90))

local hovering = false
local busyPulse = false

local function pulse()
	if busyPulse then return end
	busyPulse = true
	tween(UIStrokePF, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.45})
	tween(GlowBack, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageTransparency = 0.78})
	tween(PFPStroke, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.28})
	tween(PFPGlow, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageTransparency = 0.72})
	task.delay(0.12, function()
		if not ProfileFrame or not ProfileFrame.Parent then return end
		tween(UIStrokePF, TweenInfo.new(0.22, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.82})
		tween(GlowBack, TweenInfo.new(0.22, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageTransparency = 0.92})
		tween(PFPStroke, TweenInfo.new(0.22, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.6})
		tween(PFPGlow, TweenInfo.new(0.22, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageTransparency = 0.84})
		task.delay(0.2, function()
			busyPulse = false
		end)
	end)
end

local function hoverOn()
	hovering = true
	tween(FrameScale, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Scale = 1.05})
	tween(ProfileFrame, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Position = basePos + UDim2.fromOffset(0, -2)})
	tween(UIStrokePF, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.35})
	tween(GlowBack, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageTransparency = 0.78})
	tween(PFPStroke, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.25})
	tween(PFPGlow, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {ImageTransparency = 0.70})
	shineSweep()
	pulse()
end

local function hoverOff()

	if not (Shadow1 and Shadow1.Parent) then return end
	local t = tick()
	AuraOuterGrad.Rotation = (t * 55) % 360
	AuraInnerGrad.Rotation = (180 - (t * 75)) % 360
	SoftFillGrad.Rotation = 35 + (math.sin(t * 0.65) * 8)
	local pulseValue = (math.sin(t * 1.35) + 1) * 0.5
	local breatheValue = 0.08 + (pulseValue * 0.07)
	if not shadowHovering then
		UIStrokeS1.Transparency = 0.72 - breatheValue
		UIStrokeS2.Transparency = 0.72 - (breatheValue * 0.9)
		AuraOuter.ImageTransparency = 0.92 - (breatheValue * 0.9)
		AuraInner.ImageTransparency = 0.93 - (breatheValue * 0.8)
	end
	local targetStr = shadowHovering and 1 or 0
	shadowStrength = shadowStrength + (targetStr - shadowStrength) * math.clamp(dt * 9, 0, 1)
	if shadowStrength > 0.001 then
		local mouse = UIS:GetMouseLocation()
		local absPos = Shadow1.AbsolutePosition
		local absSize = Shadow1.AbsoluteSize
		local cx = absPos.X + absSize.X * 0.5
		local cy = absPos.Y + absSize.Y * 0.5

		local dy = (mouse.Y - cy) / 75

		Shadow2.Position = basePos2 + UDim2.fromOffset(dx * 1.15 * shadowStrength, dy * 1.15 * shadowStrength)
	end
end)

local function enableDragging(frame)
	local dragging = false
	local grabbed = false
	local dragStart
	local startPos
	local targetPos
	local velocity = Vector2.zero
	local lastPointerPos
	local trailTimer = 0
	local smoothness = 18
	local inertia = 0.88
	local velocityScale = 0.055
	local maxTilt = 4
	local trailInterval = 0.04
	local scale = frame:FindFirstChildOfClass("UIScale")
	if not scale then
		scale = Instance.new("UIScale")
		scale.Scale = 1
		scale.Parent = frame
	end
	local function beginDrag(pos)
		dragging = true
		grabbed = true
		dragStart = pos
		startPos = frame.Position
		targetPos = startPos
		velocity = Vector2.zero
		lastPointerPos = pos
		trailTimer = 0
		spawnPulse(frame, Color3.fromRGB(100, 155, 235), 0.9, 1.1, 0.25)
		TweenService:Create(scale, TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Scale = 1.03}):Play()
	end
	local function endDrag()
		if not grabbed then return end
		grabbed = false
		dragging = false
		dragStart = nil
		startPos = nil
		lastPointerPos = nil
		spawnPulse(frame, Color3.fromRGB(100, 155, 235), 1, 1.15, 0.22)
		TweenService:Create(scale, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Scale = 1}):Play()
		TweenService:Create(frame, TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Rotation = 0}):Play()
	end
	frame.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			beginDrag(input.Position)
			input.Changed:Once(function()
				if input.UserInputState == Enum.UserInputState.End then endDrag() end
			end)
		end
	end)
	UIS.InputChanged:Connect(function(input)
		if not dragging then return end
		if input.UserInputType ~= Enum.UserInputType.MouseMovement and input.UserInputType ~= Enum.UserInputType.Touch then return end
		if not dragStart or not startPos then return end
		local pos = input.Position
		local delta = pos - dragStart
		targetPos = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
		if lastPointerPos then velocity = velocity:Lerp(pos - lastPointerPos, 0.35) end
		lastPointerPos = pos
	end)
	frame.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			endDrag()
		end
	end)
	RunService.RenderStepped:Connect(function(dt)
		if not targetPos then return end
		frame.Position = frame.Position:Lerp(targetPos, math.clamp(dt * smoothness, 0, 1))
		if not dragging then
			velocity = velocity * math.pow(inertia, dt * 60)
			if velocity.Magnitude < 0.05 then
				targetPos = frame.Position
				velocity = Vector2.zero
			else
				local offset = velocity * velocityScale
				targetPos = frame.Position + UDim2.fromOffset(offset.X, offset.Y)
			end
		end
		local tilt = math.clamp(velocity.X * 0.08, -maxTilt, maxTilt)
		frame.Rotation = frame.Rotation + (tilt - frame.Rotation) * math.clamp(dt * 12, 0, 1)
		if dragging then
			trailTimer = trailTimer + dt
			if trailTimer >= trailInterval then
				trailTimer = 0
				spawnTrail(frame)
			end
		end
	end)
end

		TextTransparency = 0.15,
		TextXAlignment = Enum.TextXAlignment.Left,
		TextYAlignment = Enum.TextYAlignment.Center,
		ZIndex = 230,
	})

	local moved = 0
	local dragStart
	local startPos

	local function hoverOn()
		tw(sc, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { Scale = 1.06 })
		tw(st, TweenInfo.new(0.14), { Transparency = 0.28 })
		tw(glow, TweenInfo.new(0.14), { ImageTransparency = 0.82 })
	end

	local function hoverOff()
		if not dragging then
			tw(sc, TweenInfo.new(0.14, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { Scale = 1 })
		end
		tw(st, TweenInfo.new(0.14), { Transparency = 0.65 })
		tw(glow, TweenInfo.new(0.14), { ImageTransparency = 0.9 })
	end

	b.MouseEnter:Connect(hoverOn)
	b.MouseLeave:Connect(hoverOff)

	b.InputBegan:Connect(function(i)
		if i.UserInputType == Enum.UserInputType.MouseButton1 then
			dragging = true
			moved = 0
			dragStart = i.Position
			startPos = b.Position

			tw(sc, TweenInfo.new(0.08, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { Scale = 0.97 })

			local conn
			conn = i.Changed:Connect(function()
				if i.UserInputState == Enum.UserInputState.End then
					if conn then
						conn:Disconnect()
					end
					dragging = false
					if b:IsDescendantOf(game) then
						tw(sc, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { Scale = 1.06 })
					end
				end
			end)
		end
	end)

	b.InputChanged:Connect(function(i)
		if dragging and i.UserInputType == Enum.UserInputType.MouseMovement then
			local d = i.Position - dragStart
			moved = math.max(moved, math.abs(d.X) + math.abs(d.Y))
			b.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + d.X, startPos.Y.Scale, startPos.Y.Offset + d.Y)
		end
	end)

	b.Activated:Connect(function()
		if dragging and moved > 6 then return end

		local p = UDim2.fromScale(0.5, 0.5)
		ripple(b, p, Accent[2])
		burstRing(b, p, Accent[1])
		sparks(b, p, Accent[2])

		shine.Position = UDim2.new(0, -60, 0, 0)
		tw(shine, TweenInfo.new(0.45, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {
			Position = UDim2.new(1, 80, 0, 0),
		})

		expandUI()
	end)

	task.spawn(function()
		local i, n = 1, 2
		local breath = 0

		while b.Parent do
			breath = breath + RunService.RenderStepped:Wait()
			glow.ImageTransparency = 0.86 + (math.sin(breath * 2.2) * 0.06)
			gr.Rotation = (gr.Rotation + 0.35) % 360

			local a = Accent[i]
			local c = Accent[n]
			local t = 0

			while t < 1 and b.Parent do
				t = t + (0.9 * RunService.RenderStepped:Wait())
				local col = a:Lerp(c, math.clamp(t, 0, 1))
				st.Color = col
				dot.BackgroundColor3 = col
				glow.ImageColor3 = col
			end

			i = n
			n = (n % #Accent) + 1
		end
	end)

	return b
end

function PrestineLib:ExitFX(params)
	params = params or {}
	local root = params.Root or PrestineUI
	if not root or not root.Parent then return end

	local ExitSG = Instance.new("ScreenGui")
	ExitSG.Name = params.ScreenGuiName or "ExitSG"
	ExitSG.IgnoreGuiInset = true
	ExitSG.ResetOnSpawn = false
	ExitSG.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
	ExitSG.Parent = game:GetService("CoreGui")

	local overlay = Instance.new("Frame")
	overlay.Name = "ExitOverlay"
	overlay.BackgroundColor3 = params.OverlayColor or Color3.fromRGB(0, 0, 0)
	overlay.BackgroundTransparency = 1
	overlay.BorderSizePixel = 0
	overlay.Size = UDim2.fromScale(1, 1)
	overlay.Position = UDim2.fromScale(0.5, 0.5)
	overlay.AnchorPoint = Vector2.new(0.5, 0.5)
	overlay.ZIndex = 999999
	overlay.Parent = ExitSG

	local function fadeAll(obj, target, duration)
		local ti = TweenInfo.new(duration, Enum.EasingStyle.Quad, Enum.EasingDirection.InOut)
		for _, d in ipairs(obj:GetDescendants()) do
			if d:IsA("Frame") then
				tw(d, ti, { BackgroundTransparency = target })
			elseif d:IsA("TextLabel") or d:IsA("TextButton") or d:IsA("TextBox") then
				tw(d, ti, { TextTransparency = target })
				if d:IsA("TextBox") then
					tw(d, ti, { PlaceholderColor3 = Color3.fromRGB(60, 60, 60) })
				end
			elseif d:IsA("ImageLabel") or d:IsA("ImageButton") then
				tw(d, ti, { ImageTransparency = target })
			elseif d:IsA("UIStroke") then
				tw(d, ti, { Transparency = target })
			end
		end
	end

	local scale = root:FindFirstChild("ExitScale")
	if not scale then
		scale = Instance.new("UIScale")
		scale.Name = "ExitScale"
		scale.Scale = 1
		scale.Parent = root
	end

	local overlayDark = params.OverlayDarkness or 0.65
	local tIn = params.InTime or 0.18
	local delayMid = params.Delay or 0.16
	local tOut = params.OutTime or 0.22

	tw(overlay, TweenInfo.new(tIn, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { BackgroundTransparency = 1 - overlayDark })
	tw(scale, TweenInfo.new(tIn, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), { Scale = params.ShrinkScale or 0.98 })

	task.delay(delayMid, function()
		tw(scale, TweenInfo.new(tOut, Enum.EasingStyle.Quad, Enum.EasingDirection.In), { Scale = params.PopScale or 1.06 })
		fadeAll(root, 1, tOut)

		local t = tw(overlay, TweenInfo.new(tOut, Enum.EasingStyle.Quad, Enum.EasingDirection.In), { BackgroundTransparency = 1 })
		t.Completed:Once(function()
			if params.OnComplete then
				pcall(params.OnComplete)
			end
			if ExitSG then ExitSG:Destroy() end
			if params.DestroyRoot ~= false and root then root:Destroy() end
		end)
	end)
end

ExitButton.MouseButton1Click:Connect(function()
	PrestineLib:Confirm({
		Title = "Confirm Exit",
		Content = "Are you sure you want to close the UI?",
		Keybinds = true,
		Blur = true,
		OnConfirm = function()
			PrestineLib:ExitFX({
				Root = PrestineUI
			})
		end,
		OnCancel = function()
		end
	})
end)
