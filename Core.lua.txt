local function mk(c,p)
	local o=Instance.new(c)
	for k,v in pairs(p) do o[k]=v end
	return o
end

local function tw(o,ti,g)
	local t=TweenService:Create(o,ti,g)
	t:Play()
	return t
end

local function removeSpaces(str)
	return string.gsub(str, "%s+", "")
end

local function buttonEffectPS(button)
	local scaleDown = 0.85
	local effectFrame = Instance.new("Frame")
	effectFrame.Size = button.Size
	effectFrame.Position = UDim2.fromScale(0.5, 0.5)
	effectFrame.AnchorPoint = Vector2.new(0.5, 0.5)
	effectFrame.BackgroundColor3 = Color3.fromRGB(180, 200, 255)
	effectFrame.BackgroundTransparency = 0.95
	effectFrame.ZIndex = button.ZIndex + 1
	effectFrame.Rotation = math.random(-2, 2)
	effectFrame.Parent = button

	local corner = Instance.new("UICorner")
	corner.CornerRadius = UDim.new(0, 4)
	corner.Parent = effectFrame

	local stroke = Instance.new("UIStroke")
	stroke.Thickness = 1.5
	stroke.Transparency = 0.25
	stroke.Color = Color3.fromRGB(160, 185, 255)
	stroke.Parent = effectFrame

	local originalSize = effectFrame.Size
	local smallSize = UDim2.new(
		originalSize.X.Scale * scaleDown,
		originalSize.X.Offset * scaleDown,
		originalSize.Y.Scale * scaleDown,
		originalSize.Y.Offset * scaleDown
	)

	local bigSize = UDim2.new(
		originalSize.X.Scale * 1.05,
		originalSize.X.Offset * 1.05,
		originalSize.Y.Scale * 1.05,
		originalSize.Y.Offset * 1.05
	)

	local shrink = TweenService:Create(
		effectFrame,
		TweenInfo.new(0.08, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
		{Size = smallSize}
	)

	local pop = TweenService:Create(
		effectFrame,
		TweenInfo.new(0.12, Enum.EasingStyle.Back, Enum.EasingDirection.Out),
		{
			Size = bigSize,
			BackgroundTransparency = 0.35,
			Rotation = 0
		}
	)

	local fade = TweenService:Create(
		effectFrame,
		TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
		{
			Size = originalSize,
			BackgroundTransparency = 1
		}
	)

	local strokeTween = TweenService:Create(
		stroke,
		TweenInfo.new(0.15),
		{Transparency = 1, Thickness = 3}
	)

	shrink:Play()
	shrink.Completed:Once(function()
		pop:Play()
		strokeTween:Play()
	end)

	pop.Completed:Once(function()
		fade:Play()
	end)

	fade.Completed:Once(function()
		effectFrame:Destroy()
	end)
end

local function toggleEffectPS(MainToggle, TabGlow, isOn)
	local originalSize = MainToggle.Size
	local toggleStroke = MainToggle.Parent:FindFirstChildOfClass("UIStroke")
	local pressSize = UDim2.new(
		originalSize.X.Scale * 0.86,
		originalSize.X.Offset * 0.86,
		originalSize.Y.Scale * 0.86,
		originalSize.Y.Offset * 0.86
	)

	TweenService:Create(
		MainToggle,
		TweenInfo.new(0.06, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
		{Size = pressSize}
	):Play()

	task.delay(0.06, function()
		TweenService:Create(
			MainToggle,
			TweenInfo.new(0.18, Enum.EasingStyle.Back, Enum.EasingDirection.Out),
			{
				Size = originalSize,
				BackgroundColor3 = isOn and Color3.fromRGB(45, 75, 160) or Color3.fromRGB(30, 30, 30)
			}
		):Play()

		TweenService:Create(
			TabGlow,
			TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
			{
				Transparency = isOn and 0.2 or 0.75,
				Thickness = isOn and 3.2 or 2
			}
		):Play()

		if isOn and toggleStroke then
			local originalColor = toggleStroke.Color
			toggleStroke.Color = Color3.fromRGB(125, 175, 255)
			TweenService:Create(
				toggleStroke,
				TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
				{Color = originalColor}
			):Play()

			local pulse = Instance.new("Frame")
			pulse.AnchorPoint = Vector2.new(0.5, 0.5)
			pulse.Position = UDim2.fromScale(0.5, 0.5)
			pulse.Size = UDim2.fromScale(0.65, 0.65)
			pulse.BackgroundColor3 = Color3.fromRGB(120, 165, 255)
			pulse.BackgroundTransparency = 0.35
			pulse.ZIndex = MainToggle.ZIndex + 1
			pulse.Parent = MainToggle

			local c = Instance.new("UICorner")
			c.CornerRadius = UDim.new(0, 4)
			c.Parent = pulse

			TweenService:Create(
				pulse,
				TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
				{
					Size = UDim2.fromScale(1, 1),
					BackgroundTransparency = 1
				}
			):Play()

			task.delay(0.2, function()
				pulse:Destroy()
			end)
		end
	end)
end

local function InputEffectPS(MainInput, Stroke, state)
	local scale = MainInput:FindFirstChild("TypeScale")
	if not scale then
		scale = Instance.new("UIScale")
		scale.Name = "TypeScale"
		scale.Scale = 1
		scale.Parent = MainInput
	end

	if state == "Focus" then
		TweenService:Create(
			Stroke,
			TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
			{
				Thickness = 2.5,
				Transparency = 0.15,
				Color = Color3.fromRGB(120, 165, 255)
			}
		):Play()
		TweenService:Create(
			MainInput,
			TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
			{BackgroundColor3 = Color3.fromRGB(18, 20, 28)}
		):Play()
	elseif state == "Unfocus" then
		TweenService:Create(
			Stroke,
			TweenInfo.new(0.2, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
			{
				Thickness = 1.5,
				Transparency = 0.5,
				Color = Color3.fromRGB(140, 175, 220)
			}
		):Play()
		TweenService:Create(
			MainInput,
			TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
			{BackgroundColor3 = Color3.fromRGB(14, 16, 24)}
		):Play()
	elseif state == "Commit" then
		Stroke.Color = Color3.fromRGB(130, 180, 255)
		TweenService:Create(
			Stroke,
			TweenInfo.new(0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
			{Transparency = 0.1}
		):Play()
		task.delay(0.1, function()
			TweenService:Create(
				Stroke,
				TweenInfo.new(0.15),
				{
					Transparency = 0.5,
					Color = Color3.fromRGB(140, 175, 220)
				}
			):Play()
		end)
	elseif state == "Type" then
		Stroke.Color = Color3.fromRGB(125, 175, 255)
		Stroke.Transparency = 0.1
		TweenService:Create(
			Stroke,
			TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
			{
				Color = Color3.fromRGB(140, 175, 220),
				Transparency = 0.5
			}
		):Play()
		scale.Scale = 1
		TweenService:Create(
			scale,
			TweenInfo.new(0.06, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
			{Scale = 1.03}
		):Play()
		TweenService:Create(
			scale,
			TweenInfo.new(0.08, Enum.EasingStyle.Quad, Enum.EasingDirection.Out),
			{Scale = 1}
		):Play()
	end
end

local function SliderEffectPS(Track, Fill, Knob, state)
	local trackStroke = Track:FindFirstChild("TrackStroke")
	if not trackStroke then
		trackStroke = Instance.new("UIStroke")
		trackStroke.Name = "TrackStroke"
		trackStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
		trackStroke.Thickness = 1.5
		trackStroke.Color = Color3.fromRGB(100, 150, 255)
		trackStroke.Transparency = 0.85
		trackStroke.Parent = Track
	end

	local knobStroke = Knob:FindFirstChild("KnobStroke")
	if not knobStroke then
		knobStroke = Instance.new("UIStroke")
		knobStroke.Name = "KnobStroke"
		knobStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
		knobStroke.Thickness = 1.5
		knobStroke.Color = Color3.fromRGB(120, 165, 255)
		knobStroke.Transparency = 0.35
		knobStroke.Parent = Knob
	end

	local knobScale = Knob:FindFirstChild("KnobScale")
	if not knobScale then
		knobScale = Instance.new("UIScale")
		knobScale.Name = "KnobScale"
		knobScale.Scale = 1
		knobScale.Parent = Knob
	end

	local fillStroke = Fill:FindFirstChild("FillStroke")
	if not fillStroke then
		fillStroke = Instance.new("UIStroke")
		fillStroke.Name = "FillStroke"
		fillStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
		fillStroke.Thickness = 1.5
		fillStroke.Color = Color3.fromRGB(110, 155, 255)
		fillStroke.Transparency = 0.6
		fillStroke.Parent = Fill
	end

	local trail = Track:FindFirstChild("TrailFill")
	if state == "Down" then
		TweenService:Create(knobScale, TweenInfo.new(0.08, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Scale = 1.18}):Play()
		TweenService:Create(trackStroke, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.25, Thickness = 2.2}):Play()
		TweenService:Create(knobStroke, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.1, Thickness = 2.2, Color = Color3.fromRGB(130, 180, 255)}):Play()
		TweenService:Create(fillStroke, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.25}):Play()
		if trail then
			TweenService:Create(trail, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0.45}):Play()
			local ts = trail:FindFirstChild("TrailStroke")
			if ts then TweenService:Create(ts, TweenInfo.new(0.12, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.6}):Play() end
		end
	elseif state == "Drag" then
		TweenService:Create(knobStroke, TweenInfo.new(0.08, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Color = Color3.fromRGB(130, 180, 255), Transparency = 0.1}):Play()
	elseif state == "Up" then
		TweenService:Create(knobScale, TweenInfo.new(0.14, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Scale = 1}):Play()
		TweenService:Create(trackStroke, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.85, Thickness = 1.5}):Play()
		TweenService:Create(knobStroke, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.35, Thickness = 1.5, Color = Color3.fromRGB(120, 165, 255)}):Play()
		TweenService:Create(fillStroke, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.6}):Play()
		if trail then
			TweenService:Create(trail, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {BackgroundTransparency = 0.65}):Play()
			local ts = trail:FindFirstChild("TrailStroke")
			if ts then TweenService:Create(ts, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Transparency = 0.75}):Play() end
		end
	elseif state == "Snap" then
		local pulse = Instance.new("Frame")
		pulse.AnchorPoint = Vector2.new(0.5, 0.5)
		pulse.Position = UDim2.fromScale(0.5, 0.5)
		pulse.Size = UDim2.fromScale(0.6, 0.6)
		pulse.BackgroundColor3 = Color3.fromRGB(120, 170, 235)
		pulse.BackgroundTransparency = 0.35
		pulse.ZIndex = Knob.ZIndex + 1
		pulse.Parent = Knob
		local c = Instance.new("UICorner")
		c.CornerRadius = UDim.new(1, 0)
		c.Parent = pulse
		TweenService:Create(pulse, TweenInfo.new(0.18, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Size = UDim2.fromScale(1.25, 1.25), BackgroundTransparency = 1}):Play()
		task.delay(0.18, function()
			if pulse then pulse:Destroy() end
		end)
	end
end

local function addInteractableAnim(button, designFrame)
	local originalColor = designFrame.BackgroundColor3
	button.MouseButton1Down:Connect(function()
		local newColor = Color3.new(
			math.max(originalColor.R - 0.05, 0),
			math.max(originalColor.G - 0.05, 0),
			math.max(originalColor.B - 0.05, 0)
		)
		designFrame.BackgroundColor3 = newColor
		task.wait(0.5)
	end)
	button.MouseButton1Up:Connect(function()
		designFrame.BackgroundColor3 = originalColor
	end)
end

function PrestineLib:Set(hubName, gameName)
	if writefile then
		self.mainFolder = hubName or "UnnamedHub"
		self.gameFolder = gameName or "UnnamedGame"
		self.mainConfig = "SavedConfig.json"
		if not isfolder(self.mainFolder) then
			makefolder(self.mainFolder)
		end
		self.gameFolderPath = self.mainFolder .. "/" .. self.gameFolder
		if not isfolder(self.gameFolderPath) then
			makefolder(self.gameFolderPath)
		end
		self.filePath = self.gameFolderPath .. "/" .. self.mainConfig
		self:Load()
	end
end

function PrestineLib:Load()
	self.data = {}
	if self.filePath and isfile(self.filePath) then
		local HttpService = game:GetService("HttpService")
		local success, result = pcall(function()
			return HttpService:JSONDecode(readfile(self.filePath))
		end)
		if success and typeof(result) == "table" then
			self.data = result
		else
			warn("Failed to load existing config, using empty data.")
		end
	end
end

function PrestineLib:SetValue(key, value)
	self.data = self.data or {}
	self.data[key] = value
end

function PrestineLib:GetValue(key, fallback)
	self.data = self.data or {}
	if self.data[key] ~= nil then
		return self.data[key]
	else
		return fallback
	end
end

function PrestineLib:Save()
	if self.filePath and writefile then
		local HttpService = game:GetService("HttpService")
		local encoded = HttpService:JSONEncode(self.data)
		writefile(self.filePath, encoded)
	else
		warn("SaveManager not initialized or writefile unavailable.")
	end
end

function PrestineLib:SetGUIHotkey(key)
	local function toKeyCode(k)
		if typeof(k) == "EnumItem" and k.EnumType == Enum.KeyCode then return k end
		if typeof(k) == "string" then
			local kc = Enum.KeyCode[k]
			if kc then return kc end
		end
		return nil
	end
	local kc = toKeyCode(key)
	if not kc then return end
	self.GUIHotkey = kc
end
